(function (global) { // initializes namepsace if neccessary
  mw.namespace('mw.sdk');

  var configs = {};
  var readonlyList = {};

  /**
   *
   * @param key
   * @param value
   * @param readonly
   */
  mw.sdk.config = function (key, value, readonly) {
    if (typeof value !== 'undefined') {
      return mw.sdk.config.set(key, value, readonly);
    } else if (typeof key === 'string') {
      return mw.sdk.config.get(key);
    } else if (typeof key === 'undefined') {
      return mw.sdk.extend({}, configs);
    }
  };

  /**
   * 设置配置信息
   * @param key 配置key
   * @param value 值
   * @param readonly 是否只读
   * @returns {mw.sdk.config}
   */
  mw.sdk.config.set = function (key, value, readonly) {
    if (typeof key === 'string') {

      if (!readonlyList[key]) {
        configs[key] = value;

        if (readonly === true) {
          readonlyList[key] = true;
        }
      }

      return mw.sdk.config;
    }
  };

  mw.sdk.config.get = function (key) {
    if (key) {
      return configs[key];
    }
  };

})(this);
